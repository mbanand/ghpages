shiny.webawesome announcement prototype

Note

This is a working prototype for the eventual r-posts.com / R-bloggers announcement. The first goal is to prove that Quarto, Shinylive, webR, R-universe, and the shiny.webawesome runtime can all work together.

TL;DR

shiny.webawesome brings Web Awesome components to R Shiny through generated R wrappers, Shiny bindings, and a bundled Web Awesome runtime.

server <- function(input, output, session) {
  filtered_data <- reactive({
    if (is.null(input$species) || input$species == "all") {
      iris
    } else {
      iris[tolower(iris$Species) == input$species, , drop = FALSE]
    }
  })

  observeEvent(input$preset_sepal, {
    update_wa_select(session, "x_var", value = "Sepal.Length")
    update_wa_select(session, "y_var", value = "Sepal.Width")
  })

  observeEvent(input$preset_petal, {
    update_wa_select(session, "x_var", value = "Petal.Length")
    update_wa_select(session, "y_var", value = "Petal.Width")
  })

  observeEvent(input$open_help, {
    wa_call_method("help_drawer", "show", session = session)
  })

  observeEvent(input$close_help, {
    wa_call_method("help_drawer", "hide", session = session)
  })

  output$tab_specific_controls <- renderUI({
    active_tab <- input$surface_tabs %||% "chart"

    if (active_tab == "chart") {
      wa_container(
        class = "wa-stack",
        style = "gap: 0.85rem;",
        wa_select(
          "y_var",
          label = "Y variable",
          value = "Petal.Width",
          wa_option("Sepal length", value = "Sepal.Length"),
          wa_option("Sepal width", value = "Sepal.Width"),
          wa_option("Petal length", value = "Petal.Length"),
          wa_option("Petal width", value = "Petal.Width")
        ),
        wa_switch("show_smoother", "Show trend line", checked = TRUE)
      )
    } else if (active_tab == "summary") {
      wa_container(
        class = "wa-stack",
        style = "gap: 0.85rem;",
        wa_select(
          "focus_metric",
          label = "Summary focus",
          value = "mean",
          wa_option("Mean", value = "mean"),
          wa_option("Median", value = "median"),
          wa_option("Maximum", value = "max")
        ),
        wa_callout(
          variant = "brand",
          appearance = "soft",
          icon = "table-columns",
          "This tab keeps the same filter state and changes only how the data is summarized."
        )
      )
    } else {
      wa_container(
        class = "wa-stack",
        style = "gap: 0.85rem;",
        wa_select(
          "detail_focus",
          label = "Narrative focus",
          value = "pattern",
          wa_option("Pattern", value = "pattern"),
          wa_option("Spread", value = "spread"),
          wa_option("Recommendation", value = "recommendation")
        ),
        wa_callout(
          variant = "brand",
          appearance = "soft",
          icon = "sparkles",
          "The details tab is where explanatory copy catches up to the current filtered view."
        )
      )
    }
  })

  output$selection_badges <- renderUI({
    species_label <- if ((input$species %||% "all") == "all") {
      "All species"
    } else {
      tools::toTitleCase(input$species)
    }

    tagList(
      wa_badge(species_label, appearance = "accent"),
      wa_badge(
        sprintf(
          "%s vs %s",
          input$x_var %||% "Sepal.Length",
          input$y_var %||% "Petal.Width"
        ),
        appearance = "outlined"
      )
    )
  })

  output$tab_badge <- renderText({
    tools::toTitleCase(input$surface_tabs %||% "chart")
  })

  output$row_count <- renderText(nrow(filtered_data()))

  output$correlation_value <- renderText({
    data <- filtered_data()
    x_var <- input$x_var %||% "Sepal.Length"
    y_var <- input$y_var %||% "Petal.Width"
    sprintf("%.2f", cor(data[[x_var]], data[[y_var]]))
  })

  output$focus_value <- renderText({
    active_tab <- input$surface_tabs %||% "chart"
    if (active_tab == "summary") {
      tools::toTitleCase(input$focus_metric %||% "mean")
    } else if (active_tab == "details") {
      tools::toTitleCase(input$detail_focus %||% "pattern")
    } else {
      if (isTRUE(input$show_smoother %||% TRUE)) "Trend on" else "Trend off"
    }
  })

  output$sidebar_note <- renderText({
    active_tab <- input$surface_tabs %||% "chart"
    if (active_tab == "chart") {
      "Preset buttons jump between sepal and petal views without making you rebuild the whole plot by hand."
    } else if (active_tab == "summary") {
      "Summary focus changes the statistic, but it stays tied to the same filtered data."
    } else {
      "Narrative focus changes the interpretation copy so the demo stays visible and self-explanatory."
    }
  })

  output$iris_plot <- renderPlot({
    data <- filtered_data()
    x_var <- input$x_var %||% "Sepal.Length"
    y_var <- input$y_var %||% "Petal.Width"

    p <- ggplot(data, aes(x = .data[[x_var]], y = .data[[y_var]], color = Species)) +
      geom_point(size = 2.8, alpha = 0.85) +
      labs(x = x_var, y = y_var) +
      theme_minimal(base_size = 12) +
      theme(
        legend.position = "top",
        panel.grid.minor = element_blank(),
        panel.grid.major = element_line(color = "#d8dee9"),
        plot.background = element_rect(fill = "white", color = NA)
      )

    if (isTRUE(input$show_smoother %||% TRUE)) {
      p <- p + geom_smooth(se = FALSE, linewidth = 0.9, color = "#1d4ed8")
    }

    p
  })

  output$summary_table <- renderTable({
    data <- filtered_data()
    focus <- input$focus_metric %||% "mean"
    x_var <- input$x_var %||% "Sepal.Length"
    y_var <- input$y_var %||% "Petal.Width"

    stat_fn <- switch(
      focus,
      mean = mean,
      median = median,
      max = max
    )

    data.frame(
      measure = c(x_var, y_var),
      value = c(
        round(stat_fn(data[[x_var]]), 2),
        round(stat_fn(data[[y_var]]), 2)
      ),
      stringsAsFactors = FALSE
    )
  }, striped = TRUE, bordered = FALSE, width = "100%")

  output$details_text <- renderText({
    focus <- input$detail_focus %||% "pattern"
    x_var <- input$x_var %||% "Sepal.Length"
    y_var <- input$y_var %||% "Petal.Width"
    data <- filtered_data()
    corr <- round(cor(data[[x_var]], data[[y_var]]), 2)

    if (focus == "pattern") {
      paste(
        "The current view compares", x_var, "against", y_var, "for", nrow(data),
        "rows. The relationship is strongest when the species filter narrows the visual noise."
      )
    } else if (focus == "spread") {
      paste(
        "This filtered slice has a correlation of", corr, "between", x_var, "and",
        y_var, "which makes the separation between species either obvious or subtle depending on the chosen variables."
      )
    } else {
      "If you want a quick explanatory panel in a real app, keep it close to the chart and let the controls drive the story in plain language."
    }
  })

  output$change_text <- renderText({
    active_tab <- input$surface_tabs %||% "chart"
    if (active_tab == "chart") {
      paste(
        "The chart is using", input$x_var %||% "Sepal.Length", "on the x-axis and",
        input$y_var %||% "Petal.Width", "on the y-axis",
        if (isTRUE(input$show_smoother %||% TRUE)) "with a trend line." else "without a trend line."
      )
    } else if (active_tab == "summary") {
      paste("The summary tab is showing the", input$focus_metric %||% "mean", "for the same filtered variables.")
    } else {
      paste("The details tab is currently emphasizing", input$detail_focus %||% "pattern", "for the filtered iris subset.")
    }
  })

  output$inspector_state <- renderUI({
    species_label <- if ((input$species %||% "all") == "all") {
      "All species"
    } else {
      tools::toTitleCase(input$species)
    }

    wa_container(
      class = "wa-stack",
      style = "gap: 0.5rem;",
      wa_badge(paste("Tab:", tools::toTitleCase(input$surface_tabs %||% "chart")), appearance = "accent"),
      wa_badge(paste("Species:", species_label), appearance = "outlined"),
      wa_badge(paste("X:", input$x_var %||% "Sepal.Length"), appearance = "outlined"),
      wa_badge(paste("Y:", input$y_var %||% "Petal.Width"), appearance = "outlined")
    )
  })
}
